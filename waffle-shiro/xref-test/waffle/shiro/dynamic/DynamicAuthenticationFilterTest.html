14  package waffle.shiro.dynamic;
15  
16  import java.util.HashMap;
17  import java.util.Map;
18  
19  import javax.servlet.ServletRequest;
20  
21  import mockit.Deencapsulation;
22  
23  import org.junit.Assert;
24  import org.junit.Before;
25  import org.junit.Test;
26  import org.mockito.Matchers;
27  import org.mockito.Mockito;
28  
29  /**
30   * @author Dan Rollo Date: 2/26/13 Time: 5:47 PM
31   */
32  public class DynamicAuthenticationFilterTest {
33  
34      private static abstract class MockServletRequest implements ServletRequest {
35  
36          private final Map<String, String> parameters = new HashMap<String, String>();
37  
38          @Override
39          public String getParameter(final String name) {
40              return this.parameters.get(name);
41          }
42  
43      }
44  
45      private DynamicAuthenticationFilter dynamicAuthenticationFilter;
46  
47      private MockServletRequest          request;
48  
49      @Before
50      public void setUp() {
51          this.dynamicAuthenticationFilter = new DynamicAuthenticationFilter();
52  
53          this.request = Mockito.mock(MockServletRequest.class, Mockito.CALLS_REAL_METHODS);
54          Deencapsulation.setField(this.request, new HashMap<String, String>());
55      }
56  
57      @Test
58      public void testIsAuthTypeNegotiate() {
59          Mockito.when(this.request.getParameter(Matchers.anyString())).thenReturn(null);
60          Assert.assertFalse(this.dynamicAuthenticationFilter.isAuthTypeNegotiate(this.request));
61  
62          Mockito.when(this.request.getParameter(Matchers.anyString())).thenReturn("zzz");
63          Assert.assertFalse(this.dynamicAuthenticationFilter.isAuthTypeNegotiate(this.request));
64  
65          Mockito.when(this.request.getParameter(Matchers.anyString())).thenReturn(
66                  DynamicAuthenticationFilter.PARAM_VAL_AUTHTYPE_NEGOTIATE);
67          Assert.assertTrue(this.dynamicAuthenticationFilter.isAuthTypeNegotiate(this.request));
68      }
69  
70  }
