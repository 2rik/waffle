14  package waffle.shiro.negotiate;
15  
16  import junit.framework.TestCase;
17  import org.apache.shiro.authc.AuthenticationToken;
18  
19  /**
20   * @author Dan Rollo Date: 2/14/13 Time: 11:11 PM
21   */
22  public final class NegotiateAuthenticationRealmTest extends TestCase {
23  
24      private NegotiateAuthenticationRealm negAuthRealm;
25  
26      @Override
27      protected void setUp() throws Exception {
28          this.negAuthRealm = new NegotiateAuthenticationRealm();
29      }
30  
31      public void testSupports() {
32          assertFalse("Non-NegotiateToken should not be supported.",
33                  this.negAuthRealm.supports(new AuthenticationToken() {
34                      private static final long serialVersionUID = 334672725950031145L;
35  
36                      @Override
37                      public Object getPrincipal() {
38                          return null;
39                      }
40  
41                      @Override
42                      public Object getCredentials() {
43                          return null;
44                      }
45                  }));
46  
47          assertTrue(this.negAuthRealm.supports(new NegotiateToken(null, null, null, null, false, false, null)));
48      }
49  }
