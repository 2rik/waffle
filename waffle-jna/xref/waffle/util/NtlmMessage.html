14  package waffle.util;
15  
16  /**
17   * Rudimentary NTLM message utility.
18   * 
19   * @author dblock[at]dblock[dot]org
20   */
21  public final class NtlmMessage {
22  
23      // NTLM messages start with 0x4e544c4d53535000, NTLMSSP signature
24      private static final byte[] NTLM_SSP_SIGNATURE = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00 };
25  
26      public static boolean isNtlmMessage(final byte[] message) {
27          if (message == null || message.length < NTLM_SSP_SIGNATURE.length) {
28              return false;
29          }
30  
31          for (int i = 0; i < NTLM_SSP_SIGNATURE.length; i++) {
32              if (NTLM_SSP_SIGNATURE[i] != message[i]) {
33                  return false;
34              }
35          }
36  
37          return true;
38      }
39  
40      /**
41       * Get NTLM message type.
42       * 
43       * @param message
44       *            Assuming a valid NTLM message.
45       * @return Message type.
46       */
47      public static int getMessageType(final byte[] message) {
48          return message[NTLM_SSP_SIGNATURE.length];
49      }
50  
51      private NtlmMessage() {
52          // Prevent Instantiation of object
53      }
54  }
