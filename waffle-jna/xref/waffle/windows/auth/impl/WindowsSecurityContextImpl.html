31  /**
32   * Windows Security Context.
33   * 
34   * @author dblock[at]dblock[dot]org
35   */
36  public class WindowsSecurityContextImpl implements IWindowsSecurityContext {
37  
38      private String         principalName;
39      private String         securityPackage;
40      private SecBufferDesc  token;
41      private CtxtHandle     ctx;
42      private IntByReference attr;
43      private CredHandle     credentials;
44      private boolean        continueFlag;
45  
46      @Override
47      public IWindowsImpersonationContext impersonate() {
48          return new WindowsSecurityContextImpersonationContextImpl(this.ctx);
49      }
50  
51      @Override
52      public IWindowsIdentity getIdentity() {
53          final HANDLEByReference phContextToken = new HANDLEByReference();
54          final int rc = Secur32.INSTANCE.QuerySecurityContextToken(this.ctx, phContextToken);
55          if (WinError.SEC_E_OK != rc) {
56              throw new Win32Exception(rc);
57          }
58          return new WindowsIdentityImpl(phContextToken.getValue());
59      }
60  
61      @Override
62      public String getSecurityPackage() {
63          return this.securityPackage;
64      }
65  
66      @Override
67      public byte[] getToken() {
68          return this.token == null || this.token.getBytes() == null ? null : this.token.getBytes().clone();
69      }
70  
71      /**
72       * Get the current Windows security context for a given SSPI package.
73       * 
74       * @param securityPackage
75       *            SSPI package.
76       * @param targetName
77       *            The target of the context. The string contents are security-package specific.
78       * @return Windows security context.
79       */
80      public static IWindowsSecurityContext getCurrent(final String securityPackage, final String targetName) {
81          final IWindowsCredentialsHandle credentialsHandle = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
82          credentialsHandle.initialize();
83          try {
84              final WindowsSecurityContextImpl ctx = new WindowsSecurityContextImpl();
85              ctx.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
86              ctx.setCredentialsHandle(credentialsHandle.getHandle());
87              ctx.setSecurityPackage(securityPackage);
88              ctx.initialize(null, null, targetName);
89              return ctx;
90          } finally {
91              credentialsHandle.dispose();
92          }
93      }
94  
95      @Override
96      public void initialize(final CtxtHandle continueCtx, final SecBufferDesc continueToken, final String targetName) {
97          this.attr = new IntByReference();
98          this.ctx = new CtxtHandle();
99          int tokenSize = Sspi.MAX_TOKEN_SIZE;
100         int rc = 0;
101         do {
102             this.token = new SecBufferDesc(Sspi.SECBUFFER_TOKEN, tokenSize);
103             rc = Secur32.INSTANCE.InitializeSecurityContext(this.credentials, continueCtx, targetName,
104                     Sspi.ISC_REQ_CONNECTION, 0, Sspi.SECURITY_NATIVE_DREP, continueToken, 0, this.ctx, this.token,
105                     this.attr, null);
106             switch (rc) {
107                 case WinError.SEC_E_INSUFFICIENT_MEMORY:
108                     tokenSize += Sspi.MAX_TOKEN_SIZE;
109                     break;
110                 case WinError.SEC_I_CONTINUE_NEEDED:
111                     this.continueFlag = true;
112                     break;
113                 case WinError.SEC_E_OK:
114                     this.continueFlag = false;
115                     break;
116                 default:
117                     throw new Win32Exception(rc);
118             }
119         } while (rc == WinError.SEC_E_INSUFFICIENT_MEMORY);
120     }
121 
122     @Override
123     public void dispose() {
124         dispose(this.ctx);
125     }
126 
127     /**
128      * Dispose a security context.
129      * 
130      * @param ctx
131      *            Security context.
132      * @return True if a context was disposed.
133      */
134     public static boolean dispose(final CtxtHandle ctx) {
135         if (ctx != null && !ctx.isNull()) {
136             final int rc = Secur32.INSTANCE.DeleteSecurityContext(ctx);
137             if (WinError.SEC_E_OK != rc) {
138                 throw new Win32Exception(rc);
139             }
140             return true;
141         }
142         return false;
143     }
144 
145     @Override
146     public String getPrincipalName() {
147         return this.principalName;
148     }
149 
150     public void setPrincipalName(final String value) {
151         this.principalName = value;
152     }
153 
154     @Override
155     public CtxtHandle getHandle() {
156         return this.ctx;
157     }
158 
159     public void setCredentialsHandle(final CredHandle handle) {
160         this.credentials = handle;
161     }
162 
163     public void setToken(final byte[] bytes) {
164         this.token = new SecBufferDesc(Sspi.SECBUFFER_TOKEN, bytes);
165     }
166 
167     public void setSecurityPackage(final String value) {
168         this.securityPackage = value;
169     }
170 
171     public void setSecurityContext(final CtxtHandle phNewServerContext) {
172         this.ctx = phNewServerContext;
173     }
174 
175     @Override
176     public boolean isContinue() {
177         return this.continueFlag;
178     }
179 
180     public void setContinue(final boolean b) {
181         this.continueFlag = b;
182     }
