25  import org.junit.After;

30  import org.springframework.context.support.ClassPathXmlApplicationContext;

32  import waffle.mock.http.SimpleHttpRequest;
33  import waffle.mock.http.SimpleHttpResponse;
34  
35  /**
36   * @author dblock[at]dblock[dot]org
37   */
38  public class NegotiateSecurityFilterEntryPointTests {
39  
40      private NegotiateSecurityFilterEntryPoint entryPoint;
41      private ApplicationContext                ctx;
42  
43      @Before
44      public void setUp() {
45          String[] configFiles = new String[] { "springTestFilterBeans.xml" };
46          this.ctx = new ClassPathXmlApplicationContext(configFiles);
47          this.entryPoint = (NegotiateSecurityFilterEntryPoint) this.ctx.getBean("negotiateSecurityFilterEntryPoint");
48      }
49  
50      @After
51      public void shutDown() {
52          ((AbstractApplicationContext) this.ctx).close();
53      }
54  
55      @Test
56      public void testChallengeGET() throws IOException, ServletException {
57          SimpleHttpRequest request = new SimpleHttpRequest();
58          request.setMethod("GET");
59          SimpleHttpResponse response = new SimpleHttpResponse();
60          this.entryPoint.commence(request, response, null);
61          String[] wwwAuthenticates = response.getHeaderValues("WWW-Authenticate");
62          assertEquals(3, wwwAuthenticates.length);
63          assertEquals("NTLM", wwwAuthenticates[0]);
64          assertEquals("Negotiate", wwwAuthenticates[1]);
65          assertTrue(wwwAuthenticates[2].equals("Basic realm=\"TestRealm\""));
66          assertEquals(2, response.getHeaderNamesSize());
67          assertEquals("keep-alive", response.getHeader("Connection"));
68          assertEquals(401, response.getStatus());
69      }
70  
71      @Test(expected = ServletException.class)
72      public void testGetSetProvider() throws IOException, ServletException {
73          assertNotNull(this.entryPoint.getProvider());
74          this.entryPoint.setProvider(null);
75          SimpleHttpRequest request = new SimpleHttpRequest();
76          request.setMethod("GET");
77          SimpleHttpResponse response = new SimpleHttpResponse();
78          this.entryPoint.commence(request, response, null);
79          fail("expected ServletException");
80      }
81  }
